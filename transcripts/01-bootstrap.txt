
================================================================
  SCENARIO 1 — bootstrap a dolt repo with one table
================================================================

---- Initialize an empty dolt database (like 'git init') ----
$ dolt init --initial-branch main --name Demo --email demo@example.com
Successfully initialized dolt data repository.

---- Create a table — this is SQL, stored in dolt's own chunk format ----
$ dolt sql -q CREATE TABLE items (id INT PRIMARY KEY, name VARCHAR(64), qty INT);

---- Insert three rows ----
$ dolt sql -q INSERT INTO items VALUES (1,'apple',3),(2,'bread',1),(3,'cheese',2);

---- dolt status — uncommitted rows show as a staged/unstaged SQL-level diff ----
$ dolt status
On branch main
Untracked tables:
  (use "dolt add <table>" to include in what will be committed)
	new table:        items

---- Stage and commit the table (same two-phase model as git) ----
$ dolt add items
$ dolt commit -m seed items
commit pulta64rtn1duko68sg966pn704rh4pj (HEAD -> main) 
Author: Demo <demo@example.com>
Date:  Thu Apr 16 17:03:44 +0000 2026

	seed items


---- Inspect the history — one commit, with the same shape as git log ----
$ dolt log --oneline
pulta64rtn1duko68sg966pn704rh4pj (HEAD -> main) seed items
vubaq6aefmapfur62fkt9atk8tkcar7a Initialize data repository

---- Query it — this is just SQL, the table round-trips through the commit ----
$ dolt sql -q SELECT * FROM items;
+----+--------+-----+
| id | name   | qty |
+----+--------+-----+
| 1  | apple  | 3   |
| 2  | bread  | 1   |
| 3  | cheese | 2   |
+----+--------+-----+
